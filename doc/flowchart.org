         (lambda (x)
           (format "%s [label=\"%s\" shape=%s style=\"filled\" fillcolor=\"%s\"]"

                           (nth 1 x)

                           )) nodes "\n")

        (mapconcat
         (lambda (x)
           (concat "{rank=same "
                   (mapconcat (lambda (x) x) x " ")
                   "}"))
         rank " -> ")
        " [style=invis]\n"
        (mapconcat
         (lambda (x)
           (format "%s -> %s [taillabel=\"%s\"]"
                           (car x) (nth 1 x) (nth 2 x))) graph "\n")
        "}\n") params)
